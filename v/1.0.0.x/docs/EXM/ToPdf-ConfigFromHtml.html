Following section demonstrates how to convert a html-statement into pdf. 
using System.IO;
using Skycon.Pdf;

void main(){

    var _html= @"<html>
                    <!--PDF-DATA
                        <cfg name=""orientation"">landscape
                        <cfg name=""web.background"">true
                        <cfg name=""web.enableJavascript"">true
                        <cfg name=""web.loadImages"">true
                        <cfg name=""header.line"">true
                        <cfg name=""header.centerText"">HEADER TEXT
                        <cfg name=""footer.line"">true
                        <cfg name=""footer.leftText"">FOOTER TEXT
                        <cfg name=""footer.rightText"">Page [page]/[toPage]
                        <cfg name=""footer.fontSize"">14
                    PDF-DATA-->

                    <body>
                        <br/><br/><br/>
                        <h1>
                            <span style=""font-size:15pt;font-weight:bold;"">Font-Size 15 - (T:15)</span>
                        </h1>
                    </body>
                </html>";

    // config is generated from <!--PDF-DATA ... PDF-DATA--> block in html.
    cfg = ToPdf.CreateConfig(_html);

    var pdfData = ToPdf.Convert(_html);
}
