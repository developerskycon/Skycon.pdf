Following section demonstrates how to convert a html-statement into pdf. 
using System.IO;
using Skycon.Pdf;

void main(){

    var _html = "<Html>html-content</Html>";

    var cfg = new ToPdfConfig();
        cfg.Web.Background = true;
        cfg.Web.EnableJavascript = true;
        cfg.Web.LoadImages = true;
        cfg.IsLandscape = true;

    var pdfData = ToPdf.Convert(_html, cfg);

    using (var fs = new FileStream($"{_html}.pdf"), FileMode.Create) ) {
        fs.Write(pdfData, 0, pdfData.Length);
        fs.Close();
    }

}
