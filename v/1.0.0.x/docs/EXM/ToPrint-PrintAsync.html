Following section demonstrates how to print the pdf document in tasks as asynchronously. 
using System.IO;
using System.Threading.Tasks;
using Skycon.Pdf;

void main(){

    void fn(string taskId)
    {
        var pdfData = "byte-array data of a pdf document.";

        ToPrint.Print(pdfData);
    };

    var tasks = new List();
    for (int n = 1; n <= 50; n++)
    {
        string i = n.ToString();
        tasks.Add(Task.Factory.StartNew(() => fn(i)));
    }
    Task.WaitAll(tasks.ToArray());
}
